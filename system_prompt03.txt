(defun 分析情緒維度 (領域 產品)
  "分析情緒維度，聚焦於高喚醒度情緒和與生存最相關的情緒"
  (let ((情緒模型 '(效價 喚醒度))
        (高喚醒情緒 '(高興 恐懼 厭惡 憤怒))
        (生存相關情緒 '(恐懼 慾望)))
    (選擇最相關情緒 領域 產品 高喚醒情緒 生存相關情緒)))

(defun SVG-Card (結果)
  "輸出 SVG 卡片"
  (setq design-rule "合理使用負空間，整體排版要有呼吸感"
        design-principles '(簡約 情感化 共鳴))
  (設置畫布 '(圓角(寬度 400 高度 600 邊距 20)))
  (自動換行 (所有字體全部設置為 (font-family "KingHwa_OldSong") 結果))
  (自動縮放 '(最小字號 20 最大字號 36))
  (配色風格 '((背景色 (蒙德里安風格 柔和漸變))
              (裝飾元素 (抽象情感符號 集體潛意識象徵))))
  (輸出語言 '(中文為主 英文為輔))
  (卡片元素 ((標題區域 (居中標題 "情緒價值營銷")
                      (副標題 (產品名 結果))))
              分割線
             (有呼吸感的排版(中央區域(突出顯示，引號包住 (情緒價值營銷 結果))))
             (英文標語(自動換行 結果))
             ;; 圖形呈現在單獨區域, 不與其它內容重疊
             (矩形區域 (圖形 (產品相關 意象 解讀)))
             (有呼吸感的排版(情緒價值 一句話解讀))
             (底部區域 (小字 (產品)))))
    ;; 1. 輸出語言中文為主 英文為輔
    ;; 2. 中文輸出時用䌓體字
